\section{Hermite矩阵}
\dfn{Hermite（厄米特）矩阵}{Hermite矩阵}{
    设$n$阶复矩阵$\bm{A}$
    \begin{itemize}
        \item $\overline{\bm{A}}'=\bm{A}\Longrightarrow$Hermite矩阵
        \item $\overline{\bm{A}}'=-\bm{A}\Longrightarrow$斜Hermite矩阵.
    \end{itemize}
}
\thm{}{复矩阵分解为厄米特矩阵}{
    任一$n$阶复矩阵均可表示为一个Hermite矩阵与一个斜Hermite矩阵之和即
    \[
        \bm{A}=\frac{\bm{A}+\overline{\bm{A}}^{\prime}}{2}+
        \frac{\bm{A}-\overline{\bm{A}}^{\prime}}{2}
    \]
}
