\section{复矩阵的共轭}
\dfn{共轭}{共轭}{
    对于复矩阵，其矩阵的共轭等于元共轭的矩阵
    \[
        \overline{\bm{A}} =\left(\overline{a_{ij}} \right)_{m \times n}
    \]
}
\pro{共轭的性质}{共轭的性质}{
    \begin{enumerate}[label=（\arabic*）]
        \item $\displaystyle \overline{\left(\overline{\bm{A}}\right)}=\bm{A}$
        \item $\overline{\bm{A} + \bm{B}} =\overline{\bm{A}} +\overline{\bm{B}}  $
        \item 若$c \in \bb{C}$，则$\overline{c\bm{A}} = \overline{c}\overline{\bm{A}}$
        \item $\overline{\bm{AB}} =\overline{\bm{A}}\overline{\bm{B}}$
        \item $\overline{\left(\bm{A}^{\prime}\right)} = \left(\overline{\bm{A}}\right)^{\prime}$
        \item $\left| \overline{\bm{A}}\right|  = \overline{\left|\bm{A}\right|}$（利用组合定义非常容易证明）
    \end{enumerate}
}
